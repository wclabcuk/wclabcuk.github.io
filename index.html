class Student:

    #      값이 없음(=null)

    name = None

    age = None

    

    

    #클래스의 멤버 메소드

    # 클래스의 멤버 메소드는 클래스의 멤버 필드를 활용할 수 있는 함수

    # 함수를 만들때 무조건 매개변수 1개는 넣어야 함(꼭 self가 아녀도 됨 하지만 일반적)

    def showInfo(self):

        #클래스의 메소드 내부에서 멤버 필드에 접근하는 경우 self.멤버필드 명으로 사용

        # self 매개 변수가 사용되는 이유? 클래스의 멤버 필드들은 각각 객체마다 생성되지만, 멤버 메소드는 하나만 존재(모든 객체가 공유)

        # 이때 , 각 객체가 저장된 공간을 참조하기 위해 self 매개변수가 사용됨

        # 그래서 self 매게변수에는 현재 메소드를 실행하고 있는 객체의 참조값이 전달

        print(f"name = {self.name}, age={self.age}")

    

s1 = Student()

#클래스의 객체를 사용해 멤버 메소드를 호출할 때, 자동으로 해당 객체의 값이 전달됨

# 실제로 아래 코드가 실행되는 방식은

#Student.show_info(s1)

s1.showInfo()



s2 = Student()

s2.showInfo()
